---
title: "useOrder"
description: "Get order details by transaction hash"
---

## Usage

```ts
import { useOrder } from "@slicekit/react"

const { data: order } = useOrder({
  transactionHash,
  signature
})
```

<Note>
  See [getOrder](/core/actions/getOrder) for the corresponding action.
</Note>

## Parameters

<ParamField body="transactionHash" type="`0x${string}`" required>
  The transaction hash related to the order
</ParamField>

<ParamField body="signature" type="`0x${string}` | undefined">
  The buyer's signature required to return full shipping info. If undefined,
  incorrect or expired, partial shipping info will be returned.
</ParamField>

<ParamField body="slicerIds" type="number[] | undefined">
  Store IDs to filter `slicerOrders` by. Takes precedence over `slicerId`. If
  both undefined, all `slicerOrders` will be returned.
</ParamField>

<ParamField body="slicerId" type="number | undefined">
  Store ID to filter `slicerOrders` by. Overridden by `slicerIds` if defined. If
  both undefined, all `slicerOrders` will be returned.
</ParamField>

<ParamField body="thegraphApiKey" type="string | undefined">
  The Graph API key to use for querying the Slice subgraph. If undefined, the
  developer endpoint is used.
</ParamField>

## Return Type

```ts
import { type DetailedOrder } from "@slicekit/core"
```
